# 停止判定门控 (Stop Gate) 实现计划

> **给 agentic worker：** 使用 subagent-driven-dev 或 executing-plans 实现此计划。步骤使用 checkbox（`- [ ]`）语法跟踪。

**目标：** 在 Agent 主循环中插入停止判定门控，当模型试图停止但仍有未完成 todo 时，注入确定性 nudge 让其继续工作。

**架构：** 在 `agent.go:623` 的 `len(calls)==0` 返回前，扫描 session 中所有 `todo_write` 调用的 args，检查是否有 `pending` 或 `in_progress` 状态的任务。如有，注入固定的 turn-tail nudge 文本（常量），continue 循环。安全阀 3 次重入后强制放行。

**技术栈：** Go，现有 agent/session/provider 包，无新依赖。

**缓存安全：** nudge 文本是编译期常量，注入位置在 turn-tail（synthetic user message），不修改 L1/L2 字节。

### step-1: 创建 stop_gate.go — 核心门控逻辑

```go
package agent

import (
	"encoding/json"
	"strings"

	"tianxuan/internal/provider"
)

// stopGateReentryCap is the maximum number of times the gate can re-enter
// the loop before forcing a stop — a safety valve against a never-satisfiable
// condition burning tokens forever.
const stopGateReentryCap = 3

// stopGateNudge is the turn-tail synthetic user message injected when the model
// tries to stop but has unfinished tasks. It MUST be a compile-time constant
// to preserve DeepSeek prefix-cache stability (no per-turn variable content).
const stopGateNudge = "[system] You still have unfinished tasks — " +
	"some items in your todo list are not yet completed. " +
	"Review your todo list, continue working on the next pending item, " +
	"and only stop when every task is truly done."

// countIncompleteTodos scans session messages for todo_write calls and counts
// items whose status is NOT "completed". It reads the LAST todo_write call only
// (the model re-sends the COMPLETE list each time), so stale older snapshots
// are ignored.
func countIncompleteTodos(msgs []provider.Message) int {
	// Walk backwards to find the most recent todo_write call
	for i := len(msgs) - 1; i >= 0; i-- {
		m := msgs[i]
		if m.Role != provider.RoleAssistant {
			continue
		}
		for _, tc := range m.ToolCalls {
			if tc.Name != "todo_write" {
				continue
			}
			var p struct {
				Todos []struct {
					Status string `json:"status"`
				} `json:"todos"`
			}
			if err := json.Unmarshal([]byte(tc.Arguments), &p); err != nil {
				// Malformed args → skip this call, keep looking
				continue
			}
			incomplete := 0
			for _, t := range p.Todos {
				if strings.EqualFold(t.Status, "completed") {
					continue
				}
				incomplete++
			}
			return incomplete
		}
	}
	return 0
}

// stopGate checks whether the model's decision to stop should be overridden
// because there are unfinished tasks. Returns true when it injected a nudge
// (caller should continue the loop); false when the stop is allowed.
//
// Thread-safe: only reads session.Messages (which is written by the same
// goroutine in the run loop); the reentry counter is local to this call.
func (a *AgentRunner) stopGate() bool {
	incomplete := countIncompleteTodos(a.session.Messages)
	if incomplete == 0 {
		a.stopGateReentry = 0
		return false // no unfinished work → allow stop
	}

	a.stopGateReentry++
	if a.stopGateReentry > stopGateReentryCap {
		// Safety valve: after N re-entries, force stop so we don't loop forever
		return false
	}

	// Inject deterministic nudge as turn-tail synthetic user message.
	// Same pattern as applyStormBreaker's hint injection (agent.go:1093).
	a.session.Add(provider.Message{
		Role:    provider.RoleUser,
		Content: stopGateNudge,
	})
	return true // continue the loop
}
```

```go
package agent

import (
	"encoding/json"
	"testing"

	"tianxuan/internal/provider"
)

// makeTodoMsg 创建一个包含 todo_write tool call 的 assistant 消息。
func makeTodoMsg(statuses ...string) provider.Message {
	todos := make([]map[string]string, len(statuses))
	for i, s := range statuses {
		todos[i] = map[string]string{"content": "task " + itoa(i+1), "status": s}
	}
	args, _ := json.Marshal(map[string]any{"todos": todos})
	return provider.Message{
		Role: provider.RoleAssistant,
		ToolCalls: []provider.ToolCall{{
			ID:        "call_1",
			Name:      "todo_write",
			Arguments: string(args),
		}},
	}
}

// TestCountIncompleteTodosAllDone：全部 completed → 返回 0
func TestCountIncompleteTodosAllDone(t *testing.T) {
	msgs := []provider.Message{
		{Role: provider.RoleSystem, Content: "system"},
		{Role: provider.RoleUser, Content: "hello"},
		makeTodoMsg("completed", "completed"),
	}
	n := countIncompleteTodos(msgs)
	if n != 0 {
		t.Fatalf("expected 0 incomplete, got %d", n)
	}
}

// TestCountIncompleteTodosMixed：混合状态 → 只数非 completed
func TestCountIncompleteTodosMixed(t *testing.T) {
	msgs := []provider.Message{
		makeTodoMsg("completed", "in_progress", "pending"),
	}
	n := countIncompleteTodos(msgs)
	if n != 2 {
		t.Fatalf("expected 2 incomplete, got %d", n)
	}
}

// TestCountIncompleteTodosLastWins：只读最后一个 todo_write
func TestCountIncompleteTodosLastWins(t *testing.T) {
	msgs := []provider.Message{
		makeTodoMsg("pending", "pending", "pending"), // 旧的，忽略
		makeTodoMsg("completed", "completed"),         // 最新的：全部完成
	}
	n := countIncompleteTodos(msgs)
	if n != 0 {
		t.Fatalf("expected 0 (last wins), got %d", n)
	}
}

// TestCountIncompleteTodosNoCall：没有 todo_write → 返回 0
func TestCountIncompleteTodosNoCall(t *testing.T) {
	msgs := []provider.Message{
		{Role: provider.RoleUser, Content: "hello"},
	}
	n := countIncompleteTodos(msgs)
	if n != 0 {
		t.Fatalf("expected 0, got %d", n)
	}
}

// TestStopGateAllowsWhenDone：无未完成 → stopGate 返回 false，不注入消息
func TestStopGateAllowsWhenDone(t *testing.T) {
	s := NewSession("")
	s.Add(makeTodoMsg("completed"))
	a := &AgentRunner{session: s}
	before := len(s.Messages)
	blocked := a.stopGate()
	if blocked {
		t.Fatal("expected false (allow stop), got true")
	}
	if len(s.Messages) != before {
		t.Fatalf("expected no new messages, got +%d", len(s.Messages)-before)
	}
}

// TestStopGateBlocksWhenPending：有未完成 → stopGate 返回 true，注入 nudge
func TestStopGateBlocksWhenPending(t *testing.T) {
	s := NewSession("")
	s.Add(makeTodoMsg("pending", "completed"))
	a := &AgentRunner{session: s}
	blocked := a.stopGate()
	if !blocked {
		t.Fatal("expected true (block stop), got false")
	}
	last := s.Messages[len(s.Messages)-1]
	if last.Role != provider.RoleUser {
		t.Fatalf("expected user-role nudge, got %s", last.Role)
	}
	if last.Content != stopGateNudge {
		t.Fatalf("expected stopGateNudge constant, got %q", last.Content)
	}
}

// TestStopGateSafetyValve：3 次重入后强制放行
func TestStopGateSafetyValve(t *testing.T) {
	s := NewSession("")
	s.Add(makeTodoMsg("pending"))
	a := &AgentRunner{session: s}

	// 前 3 次应该阻塞
	for i := 0; i < stopGateReentryCap; i++ {
		if !a.stopGate() {
			t.Fatalf("reentry %d: expected true (block)", i+1)
		}
	}
	// 第 4 次应该放行
	if a.stopGate() {
		t.Fatal("after cap: expected false (allow), got true")
	}
}

// TestStopGateNudgeTextIsConstant：验证 nudge 文本是编译期常量
func TestStopGateNudgeTextIsConstant(t *testing.T) {
	// 如果有人在运行时拼接字符串到 nudge 中，这个测试会失败
	if stopGateNudge == "" {
		t.Fatal("stopGateNudge is empty")
	}
	// 确保不包含格式化动词
	if contains(stopGateNudge, "%") {
		t.Fatal("stopGateNudge contains %% — must be a compile-time constant with no format verbs")
	}
}

func contains(s, substr string) bool {
	for i := 0; i <= len(s)-len(substr); i++ {
		if s[i:i+len(substr)] == substr {
			return true
		}
	}
	return false
}
```
