# V1.5 三层缓存重构 — 实现计划

> **给 agentic worker：** 可选执行方式：使用 superpowers:subagent-driven-dev（推荐）或 superpowers:executing-plans。使用 checkbox（`- [ ]`）跟踪进度。

**目标：** 将 system prompt 拆为 L1（不可变前缀）和 L2（首轮锁定运行时上下文），消除 compiler 可变状态，FailureHint 改为 turn-tail 注入，目标缓存命中率 ≥99.3%。

**架构：** L1 = compiler.SystemPrompt() 全程不变；L2 = RuntimeContext 作为 messages[1] 独立注入，首轮后锁定；L3 = session.Messages[2:] 对话追加不变。

**技术栈：** Go 1.24+，纯标准库 + 现有项目依赖。

**回滚锚点：** `git checkout v1.4.0 -b rollback-v1.4`

### 步骤 2：实现 RuntimeContext（TDD GREEN）

**目标：** 最小实现让测试通过。

- [ ] 创建 `internal/cache/runtime.go`

```go
package cache

import (
	"strings"
	"sync"
)

// RuntimeContext holds the L2 context that is injected as a separate system
// message between the L1 prefix and the conversation history. It is assembled
// once during the first turn (profile + route result) and then locked so
// DeepSeek's prefix cache can serve it on subsequent turns.
type RuntimeContext struct {
	mu           sync.Mutex
	profile      string
	taskKind     string
	contextFocus string
	hints        []string
	locked       bool
}

// NewRuntimeContext creates an empty, unlocked RuntimeContext.
func NewRuntimeContext() *RuntimeContext {
	return &RuntimeContext{}
}

// InjectProfile stores the project profile text. Ignored when locked.
func (rc *RuntimeContext) InjectProfile(profile string) {
	rc.mu.Lock()
	defer rc.mu.Unlock()
	if rc.locked {
		return
	}
	rc.profile = profile
}

// ApplyRoute records the GoalRouter classification result. Ignored when locked.
func (rc *RuntimeContext) ApplyRoute(cfg DomainConfig) {
	rc.mu.Lock()
	defer rc.mu.Unlock()
	if rc.locked {
		return
	}
	if cfg.Kind != "" && cfg.Kind != KindDefault {
		rc.taskKind = string(cfg.Kind)
	}
	if len(cfg.ContextFocus) > 0 {
		var b strings.Builder
		b.WriteString("## Focus Areas\n\n")
		b.WriteString("The user's request involves these project areas:\n\n")
		for _, f := range cfg.ContextFocus {
			b.WriteString("- `")
			b.WriteString(f)
			b.WriteString("`\n")
		}
		rc.contextFocus = b.String()
	}
}

// AppendHint adds a corrective hint from the FailureDetector. Allowed after lock
// — this is how the L2 can adapt across turns while keeping the core stable.
func (rc *RuntimeContext) AppendHint(hint string) {
	if hint == "" {
		return
	}
	rc.mu.Lock()
	defer rc.mu.Unlock()
	rc.hints = append(rc.hints, hint)
}

// Lock marks the context as immutable for profile/route changes. Hints can still
// be appended. Idempotent.
func (rc *RuntimeContext) Lock() {
	rc.mu.Lock()
	defer rc.mu.Unlock()
	rc.locked = true
}

// IsLocked reports whether Lock() has been called.
func (rc *RuntimeContext) IsLocked() bool {
	rc.mu.Lock()
	defer rc.mu.Unlock()
	return rc.locked
}

// SystemPrompt returns the L2 content as a single string suitable for a system
// message. Returns "" when no content has been injected (first turn, no overhead).
func (rc *RuntimeContext) SystemPrompt() string {
	rc.mu.Lock()
	defer rc.mu.Unlock()
	var parts []string
	if rc.profile != "" {
		parts = append(parts, rc.profile)
	}
	if rc.taskKind != "" {
		parts = append(parts, "Task type: "+rc.taskKind)
	}
	if rc.contextFocus != "" {
		parts = append(parts, rc.contextFocus)
	}
	for _, h := range rc.hints {
		parts = append(parts, "[Hint: "+h+"]")
	}
	return strings.Join(parts, "\n\n")
}
```

```go
func (a *Agent) stream(ctx context.Context, turn int) (string, string, string, []provider.ToolCall, *provider.Usage, error) {
	tools := a.tools.Schemas()
	a.activeSchemasMu.RLock()
	if a.activeSchemas != nil {
		tools = a.activeSchemas
	}
	a.activeSchemasMu.RUnlock()
	
	// Build messages with L2 injection: [L1 system] [L2 runtime] [history...]
	msgs := a.session.Messages
	if a.runtimePrompt != "" {
		// Insert L2 as a second system message right after the first (L1).
		// a.session.Messages[0] is always the L1 system prompt.
		injected := make([]provider.Message, 0, len(msgs)+1)
		injected = append(injected, msgs[0]) // L1
		injected = append(injected, provider.Message{Role: provider.RoleSystem, Content: a.runtimePrompt}) // L2
		injected = append(injected, msgs[1:]...) // conversation history
		msgs = injected
	}
	
	ch, err := a.prov.Stream(ctx, provider.Request{
		Messages:    msgs,
		Tools:       tools,
		Temperature: a.temperature,
	})
	// ... rest unchanged
```
